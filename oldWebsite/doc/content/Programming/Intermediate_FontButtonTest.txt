import java.awt.*;<br>
import java.awt.event.*;<br>
import javax.swing.*;<br>

//this is the driver that handels dealing with the frames<br>
//it creates the FontButtonFrame which uses the other pannels<br>
public class FontButtonTest<br>
{<br>
    public static void main (String[] args)<br>
    {<br>
	FontButtonFrame frame = new FontButtonFrame ();<br>
	frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE);<br>
	frame.show ();<br>
    } // end method main<br>
} // end class ButtonTest<br>

// A frame with a button panel<br>
//This methoc creates a frame that is to be opened in the center<br>
//of the screen. It also creates a main pannel inside the frame<br>
//that deals with the other pannels<br>
class FontButtonFrame extends JFrame<br>
{<br>
    public FontButtonFrame ()<br>
    {<br>
	setTitle ("FontButtonTest");<br>
	Toolkit kit = Toolkit.getDefaultToolkit ();<br>
	Dimension screenSize = kit.getScreenSize ();<br>
	int screenWidth = screenSize.width;<br>
	int screenHeight = screenSize.height;<br>
	Image img = kit.getImage ("Button.jpg");<br>
	setIconImage (img);<br>

	setSize (screenWidth / 2, screenHeight / 2);<br>
	setLocation (screenWidth / 4, screenHeight / 4);<br>

	Container contentPane = getContentPane ();<br>
	MainPanel panel = new MainPanel ();<br>
	contentPane.add (panel);<br>
    }<br>
} // end class ButtonFrame<br>

//the MainPannel handels dealing with the FlagFont and FontPanel so that<br>
//the FontButtonFrame can distuinguish between the two pannels.<br>
class MainPanel extends JPanel<br>
{<br>
    public MainPanel ()<br>
    {<br>
	setLayout (new BorderLayout ());<br>
	// Set up the FacePanel<br>
	FontPanel font = new FontPanel ();<br>
	add (font, BorderLayout.CENTER);<br>
	// Set up the buttonPanel<br>
	FlagFont panel = new FlagFont (font);<br>
	add (panel, BorderLayout.SOUTH);<br>
    } // end constructor<br>
} // end class MainPanel<br>

//this pannel creates the buttens that are used to change the color of<br>
//the font accordingly. this pannel only performs the action and does<br>
//not take part in listening to any action to perform it.<br>
class FlagFont extends JPanel<br>
{<br>
    public FlagFont (ActionListener al)<br>
    {<br>
	setBackground (Color.lightGray);<br>
	//sets the background of the button to the given collor<br>
	setLayout (new FlowLayout ()); //sets the layout of the button<br>
	// Create and add red button<br>
	JButton btnRed = new JButton ("Red");<br>
	btnRed.setBackground (Color.red);<br>
	btnRed.addActionListener (al);<br>
	add (btnRed);<br>
	// Create and add white button<br>
	JButton btnWhite = new JButton ("White");<br>
	btnWhite.setBackground (Color.white);<br>
	btnWhite.addActionListener (al);<br>
	add (btnWhite);<br>
	// Create and add blue button<br>
	JButton btnBlue = new JButton ("Blue");<br>
	btnBlue.setBackground (Color.blue);<br>
	btnBlue.addActionListener (al);<br>
	add (btnBlue);<br>
    } // end constructor<br>
} // end class FlagFont<br>

//This pannel creates a word, sets the background color, and then performs<br>
//the action that the FlagFont creates. This pannel does all of the Action<br>
//listening.<br>
class FontPanel extends JPanel implements ActionListener<br>
{<br>
    Color fontColor = Color.green;<br>
    public void paintComponent (Graphics g)<br>
    {<br>
	super.paintComponent (g);<br>
<br>
	g.setColor (fontColor);<br>
	g.setFont (new Font ("Comic Sans MS", Font.BOLD + Font.ITALIC, 36));<br>
	g.drawString ("muckluck", 160, 180);<br>

    }<br>


    public FontPanel ()<br>
    {<br>
	setBackground (Color.lightGray);<br>
    }<br>


    public void actionPerformed (ActionEvent event)<br>
    {<br>
	String cmd = event.getActionCommand ();<br>
	//long i;<br>
	if (cmd.equals ("Red"))<br>
	{<br>
	    fontColor = Color.red;<br>
	    repaint ();<br>
	}<br>
	else if (cmd.equals ("White"))<br>
	{<br>
	    fontColor = Color.white;<br>
	    repaint ();<br>
	}<br>
	else if (cmd.equals ("Blue"))<br>
	{<br>
	    fontColor = Color.blue;<br>
	    repaint ();<br>
	}<br>
	else<br>
	    System.out.println ("Unexpected Error!");<br>
    }<br>
} // end class ShapesPanel<br>
